115inline void QuaternionMultiply(double res[4], const double p[4],
116 const double q[4]) {
117 double p_vec[3]{p[1], p[2], p[3]};
118 double q_vec[3]{q[1], q[2], q[3]};
119 double p_cross_q[3];
120 VectorCross(p_cross_q, p_vec, q_vec);
121 res[0] = p[0] * q[0] - VectorDot(p_vec, q_vec);
122 res[1] = p[0] * q_vec[0] + q[0] * p_vec[0] + p_cross_q[0];
123 res[2] = p[0] * q_vec[1] + q[0] * p_vec[1] + p_cross_q[1];
124 res[3] = p[0] * q_vec[2] + q[0] * p_vec[2] + p_cross_q[2];
125}

149inline void RotateByRodrigues(double val_new[3], const double val_old[3],
150 double rot_angle_cos, double rot_angle_sin,
151 const double rot_axis[3]) {
152 double by_axis_norm = 1.0 / norm(rot_axis);
153 double rot_axis_unit[3]{rot_axis[0] * by_axis_norm,
154 rot_axis[1] * by_axis_norm,
155 rot_axis[2] * by_axis_norm};
156
157 double axis_cross_val[3];
158 VectorCross(axis_cross_val, rot_axis_unit, val_old);
159 double axis_dot_val = VectorDot(rot_axis_unit, val_old);
160
161 val_new[0] = val_old[0] * rot_angle_cos + axis_cross_val[0] * rot_angle_sin +
162 rot_axis_unit[0] * axis_dot_val * (1.0 - rot_angle_cos);
163 val_new[1] = val_old[1] * rot_angle_cos + axis_cross_val[1] * rot_angle_sin +
164 rot_axis_unit[1] * axis_dot_val * (1.0 - rot_angle_cos);
165 val_new[2] = val_old[2] * rot_angle_cos + axis_cross_val[2] * rot_angle_sin +
166 rot_axis_unit[2] * axis_dot_val * (1.0 - rot_angle_cos);
